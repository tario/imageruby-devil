  def create_temp_file(basename, data)
    tmpfile = Tempfile.new(basename)
    ret = nil
    begin
      tmpfile.write data
      tmpfile.close
      ret = yield(tmpfile)
    ensure
      tmpfile.delete
    end

    ret
  end

  end
