      def to_devil
        path = ImageRubyDevilMixin.tmppath+".bmp"
        save(path,:bmp)

        use_temp_file(path) do
          if block_given?
            Devil.with_image(path) do |devil_image|
              yield(devil_image)
            end
          else
            Devil.load_image(path)
          end
        end
      end
